# One rule per line. First match wins within each audience.
# Format:
# audience=<audience>|rule_name=<name>|condition=<pandas.eval expression>|segment=<segment>|action=<action>|recommended_channel=<channel>|priority=<priority>
#
# Rule design notes:
# - Keep the model score as the main ranking signal.
# - Only add a small set of support variables that stayed useful in the holdout.
# - Churn support variables: is_near_graduation plus channel eligibility.
# - Re-engagement support variables: days_since_last_activity, has_ever_redeemed, scan_count_60d, plus channel eligibility.
# - Thresholds were picked after checking holdout behavior:
#   - churn_30_to_60_prob >= 0.75 marks the strongest churn-risk pocket
#   - re_engage_30d_prob >= 0.60 marks a hot audience
#   - re_engage_30d_prob >= 0.40 marks a warm audience

# Why this exists: high churn score plus near graduation usually looks more like a natural lifecycle transition than a preventable drop-off, so the action stays light and low-cost.
audience=churn|rule_name=transition_guardrail_push|condition=churn_30_to_60_prob >= 0.75 & is_near_graduation == 1 & channel_eligible_phone_push == 1|segment=physiological_transition|action=low_roi_transition_guardrail|recommended_channel=push|priority=medium

# Why this exists: high churn score away from graduation is the clearest preventable-risk case, and push is the fastest high-priority channel when it is available.
audience=churn|rule_name=preventable_push|condition=churn_30_to_60_prob >= 0.75 & is_near_graduation == 0 & channel_eligible_phone_push == 1|segment=preventable_churn|action=preventable_churn_core|recommended_channel=push|priority=high

# Why this exists: this is the same preventable high-risk case, but for users who cannot be reached by push and can still be reached by email.
audience=churn|rule_name=preventable_email|condition=churn_30_to_60_prob >= 0.75 & is_near_graduation == 0 & channel_eligible_email == 1|segment=preventable_churn|action=preventable_churn_core|recommended_channel=email|priority=high

# Why this exists: medium churn scores are still risky on the holdout, but they do not justify the same urgency as the top-risk band, so they get a softer save action.
audience=churn|rule_name=watchlist_email|condition=churn_30_to_60_prob >= 0.60 & channel_eligible_email == 1|segment=watchlist_churn|action=soft_save_nudge|recommended_channel=email|priority=medium

# Why this exists: some users score as risky but cannot be contacted, so the engine should surface the risk without pretending there is an outbound action.
audience=churn|rule_name=high_risk_no_channel|condition=churn_30_to_60_prob >= 0.75|segment=high_risk_no_channel|action=no_action|recommended_channel=none|priority=low

# Why this exists: everyone else falls back to monitoring only, which keeps the churn side simple and avoids over-targeting weak cases.
audience=churn|rule_name=default_churn|condition=True|segment=monitor_only|action=no_action|recommended_channel=none|priority=low

# Why this exists: a high re-engagement score plus activity in the last two weeks is the clearest hot-user pattern, so it gets the strongest mission-led action when push is available.
audience=engagement|rule_name=hot_user_push|condition=re_engage_30d_prob >= 0.60 & days_since_last_activity <= 14 & channel_eligible_phone_push == 1|segment=hot_user|action=mission_bundle_nudge|recommended_channel=push|priority=high

# Why this exists: this covers the same hot-user pattern when push is not available but email still is.
audience=engagement|rule_name=hot_user_email|condition=re_engage_30d_prob >= 0.60 & days_since_last_activity <= 14 & channel_eligible_email == 1|segment=hot_user|action=mission_bundle_nudge|recommended_channel=email|priority=high

# Why this exists: among warm users, prior redemption is a simple trust signal that makes reward reminders more credible than generic content.
audience=engagement|rule_name=warm_loyalty_email|condition=re_engage_30d_prob >= 0.40 & has_ever_redeemed == 1 & channel_eligible_email == 1|segment=warm_user|action=reward_reminder|recommended_channel=email|priority=medium

# Why this exists: scan-heavy users are still showing recent loyalty behavior, so a content or mission nudge is a good medium-priority follow-up when push is available.
audience=engagement|rule_name=warm_scanner_push|condition=re_engage_30d_prob >= 0.40 & scan_count_60d >= 2 & channel_eligible_phone_push == 1|segment=warm_user|action=content_mission_nurture|recommended_channel=push|priority=medium

# Why this exists: users with no usable channel should be labeled clearly so the engine does not suggest a campaign that cannot be executed.
audience=engagement|rule_name=not_eligible|condition=channel_eligible_email == 0 & channel_eligible_phone_push == 0|segment=not_campaign_eligible|action=no_action|recommended_channel=none|priority=low

# Why this exists: everyone else stays in a light-touch watchlist bucket, which keeps the engagement side broad but still controlled.
audience=engagement|rule_name=default_engagement|condition=True|segment=watchlist|action=light_touch_nurture|recommended_channel=email|priority=low
